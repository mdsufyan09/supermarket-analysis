pipeline {
    agent any
    
    stages {
        stage('Setup Python Environment') {
            steps {
                // Install required dependencies (ensure pip and python are available)
                sh 'pip install pandas matplotlib seaborn'
            }
        }
        
        stage('Checkout Code') {
            steps {
                // Pull the latest code from the repository
                git url: 'https://github.com/your-repo/supermarket_analysis.git', branch: 'main'
            }
        }

        stage('Run Analysis') {
            steps {
                // Run the Python analysis script
                sh 'python supermarket_analysis.py'
            }
        }
    }

    post {
        always {
            // Archive artifacts like reports or images generated by the script
            archiveArtifacts artifacts: '/.png, **/.csv'
        }
    }
}
